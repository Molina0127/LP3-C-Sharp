	public class Program
	{

		{
			bool num = true;
			int numero, i;
			
			do{

				numero = int.Parse(Console.ReadLine());
					
				if (numero % 2 == 0 && numero > 0)

					num = false;
				}

			Console.WriteLine("\n//NÃºmeros Pares de 0 a "+numero+"//");

			}
		}
